// hsv.vs.txt (sm4)

in vec3 a_position;
in vec3 a_normal;
in vec4 a_color;
in vec2 a_texcoord;

out vec3  v_normal;
out vec4  v_color;
out vec2  v_texcoord;

uniform mat4 model_to_clip;

void main()
{
    gl_Position = gui.model_to_clip * vec4(a_position.xyz, 1.0);
    v_normal    = a_normal;
    v_color     = a_color;
    v_texcoord  = a_texcoord;
}
