// slider.fs.txt (sm4)

in  vec2    v_texcoord;

out vec4    out_color;

void main(void)
{
    vec4 c = texture(background_texture, v_texcoord.xy);
    out_color = (gui.color_scale * c); // + gui.color_add * c.a;

    if (gl_FragCoord.x <= gui.t)
    {
        out_color.rgb += c.a * gui.color_add.rgb;
    }
}
