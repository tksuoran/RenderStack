// basic.fs.txt (sm4)

in vec3 v_normal;
in vec4 v_color;
in vec2 v_texcoord;

out vec4 out_color;

void main(void)
{
    vec3 L = vec3(0.0, 1.0, 0.0);
    vec3 N = normalize(v_normal);
    float ln = dot(L, N);
    float ln_clamped = clamp(ln, 0.0, 1.0);
    vec4 color = pow(ln_clamped, 1.0/2.2) * v_color;
    out_color = vec4(color.xyz, v_color.a);
    out_color.rgb = global.color.rgb * (vec3(0.5) + 0.5 * N);
}
